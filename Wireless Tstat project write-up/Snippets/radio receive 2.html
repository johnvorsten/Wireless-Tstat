void ReceiveData() {
  //Standard Receive procedure for VirtualWire
  if (vw_get_message(buf, &buflen)) {
    int i;
    Serial.print("got: ");
    for ( i = 0; i < buflen; i++) {
      Serial.print(buf[i], DEC);
      Serial.print(" ");
    }
    Serial.println();
  } else {
    Serial.println(F("No Data Received"));
  }

  //######## Convert RF Data to floating ########
  //We receive the data in a buffer of 20 bytes; but the original
  //data was 5 floating point numbers.  We need to convert the 20 bytes
  //to 5 floating point numbers.

  //First we need to definte the locations for untions to be created and
  for (int i = 0; i < 4; i++) {
    bytes4[i] = buf[i];}
  for (int i = 4; i < 8; i++) {
    bytes5[i - 4] = buf[i];}
  for (int i = 8; i < 12; i++) {
    bytes6[i - 8] = buf[i];}
  for (int i = 12; i < 16; i++) {
    bytes7[i - 12] = buf[i];}
  for (int i = 16; i < 20; i++) {
    bytes8[i - 16] = buf[i];}

  //call the memcpy function. Location of received bytes will be
  //a set location in the Temp_Store array
  memcpy(&Temp_Store[4], &bytes4, sizeof(Temp_Store[0]));
  memcpy(&Temp_Store[5], &bytes5, sizeof(Temp_Store[0]));
  memcpy(&Temp_Store[6], &bytes6, sizeof(Temp_Store[0]));
  memcpy(&Temp_Store[7], &bytes7, sizeof(Temp_Store[0]));
  memcpy(&Temp_Store[8], &bytes8, sizeof(Temp_Store[0]));
}
