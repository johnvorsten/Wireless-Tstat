//######## Thermistor calculation function ########
float GetTemp(float R_Thermistor, int ResArray[38], int Temp_Array[38], int Thermistor_Pin)  {
  int i = 0;
  Thermistor_Voltage = analogRead(Thermistor_Pin) / 1023.00 * Input_Voltage;
  R_Thermistor = Thermistor_Voltage * R_Known / (Input_Voltage - Thermistor_Voltage);
  while (R_Thermistor < Resistance_Array[i]) {
    i++;
  }
  Temperature = (float)(R_Thermistor - ResArray[i - 1]) / (ResArray[i] - ResArray[i - 1]) * (Temp_Array[i] - Temp_Array[i - 1]) + Temp_Array[i - 1];
  return Temperature;
}
